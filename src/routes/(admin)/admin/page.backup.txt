<script lang="ts">
	import { onMount } from 'svelte';

	let stats = $state({
		totalUsers: 0,
		totalCargo: 0,
		pendingCargo: 0,
		totalRevenue: 0,
		activeOfficers: 0,
		deliveredToday: 0,
		inTransit: 0,
		readyForShipping: 0
	});

	let recentPackages = $state([]);
	let isLoading = $state(true);

	onMount(async () => {
		await loadDashboardData();
	});

	async function loadDashboardData() {
		isLoading = true;
		try {
			// Load stats
			const statsResponse = await fetch('/api/admin/stats', {
				credentials: 'include'
			});

			if (statsResponse.ok) {
				const statsData = await statsResponse.json();
				stats = {
					totalUsers: statsData.totalUsers || 0,
					totalCargo: statsData.totalCargo || 0,
					pendingCargo: statsData.pendingCargo || 0,
					totalRevenue: statsData.totalRevenue || 0,
					activeOfficers: statsData.activeOfficers || 0,
					deliveredToday: 0,
					inTransit: 0,
					readyForShipping: 0
				};
			}

			// Load recent packages
			const cargoResponse = await fetch('/api/cargo?limit=5', {
				credentials: 'include'
			});

			if (cargoResponse.ok) {
				const cargoData = await cargoResponse.json();
				recentPackages = cargoData.cargo || [];
			}
		} catch (error) {
			console.error('Failed to load dashboard data:', error);
		} finally {
			isLoading = false;
		}
	}

	function getStatusColor(status: string): string {
		const colors: Record<string, string> = {
			'Prepared': '#F59E0B',
			'Ready for shipping': '#3B82F6',
			'In transit': '#8B5CF6',
			'Delivered': '#22C55E',
			'Pending': '#EF4444'
		};
		return colors[status] || '#6B7280';
	}
</script>
